429 std::string LogReaderFactory::Locate(const std::string& mcFile) const {
430  TString inFileName(mcFile);
431  TString logFileName;
432  std::string extension;
433  if (mcFile.find_last_of('.') != std::string::npos) {
434  extension = mcFile.substr(mcFile.find_last_of('.'));
435  } // if
436  // If the input file is in the current directory, expand the full path:
437  if (std::string(".") == gSystem->DirName(inFileName)) {
438  inFileName.Prepend("/").Prepend(gSystem->pwd());
439  } // if
440 
441  // The standard data directory structure is to have a directory called
442  // TXTFILES containing the Monte Carlo output, and a directory called
443  // LOGFILES containing the corresponding log files. The sub-directory
444  // structure of LOGFILES should match that of TXTFILES.
445  // So, first we'll check if the input path contains TXTFILES, in which
446  // case we just substitute LOGFILES:
447  if (inFileName.Contains("TXTFILES")) {
448  logFileName = inFileName;
449  logFileName.ReplaceAll("TXTFILES", "LOGFILES");
450  logFileName.ReplaceAll(extension.c_str(), ".log");
451  } // if...
452  // Check if the file whose name we have constructed exists.
453  // If not clear the string contents.
454  if (gSystem->AccessPathName(logFileName, kFileExists)) {
455  logFileName.Clear();
456  } // if
457  // OK, that didn't work, so let's just look in the current directory
458  if (logFileName.IsNull()) {
459  logFileName = inFileName;
460  if (extension.empty()) {
461  logFileName.Append(".log");
462  } else {
463  logFileName.ReplaceAll(extension.c_str(), ".log");
464  } // if
465  } // if
466  // Check if the file whose name we have constructed exists.
467  // If not clear the string contents.
468  if (gSystem->AccessPathName(logFileName, kFileExists)) {
469  logFileName.Clear();
470  } // if
471  return logFileName.Data();
472 }
